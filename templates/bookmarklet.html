২ Manual Method (Mobile):

প্রথমে এই page টা bookmark করো। তারপর bookmark edit করে URL এর জায়গায় নিচের code paste করো:
javascript:(function(){var APP='https://anisubv2-production.up.railway.app';var urls=[];var found=[];Array.from(document.querySelectorAll('source,video')).forEach(function(e){var s=e.src||e.getAttribute('src')||'';if(s)urls.push(s);});Array.from(document.querySelectorAll('iframe')).forEach(function(e){var s=e.src||e.getAttribute('src')||'';if(s&&s.startsWith('http'))urls.push(s);});var scripts=document.body.innerHTML;var m3u8s=scripts.match(/https?:\/\/[^\s"'<>]+\.m3u8[^\s"'<>]*/g)||[];urls=urls.concat(m3u8s);urls=urls.filter(function(u,i){return urls.indexOf(u)===i;});if(urls.length===0){alert('কোনো video URL পাওয়া যায়নি। page টি পুরো load হলে আবার try করুন।');return;}var list=urls.map(function(u,i){return(i+1)+'. '+u.substring(0,80)+'...'}).join('\n');var choice=prompt('পাওয়া গেছে '+urls.length+' টি URL:\n\n'+list+'\n\nnumber দিন (1-'+urls.length+'):','1');if(!choice)return;var url=urls[parseInt(choice)-1];if(!url){alert('ভুল নম্বর!');return;}window.open(APP+'?video_url='+encodeURIComponent(url),'_blank');})();
৩ যেকোনো episode page এ গিয়ে সেই bookmark open করো — video URL automatically বের হবে!
